// ============ DATA LAYER ============
function getStores() {
  return JSON.parse(localStorage.getItem('mtjri_stores') || '{}');
}
function saveStores(data) {
  localStorage.setItem('mtjri_stores', JSON.stringify(data));
}
function getOrders(storeKey) {
  return JSON.parse(localStorage.getItem('mtjri_orders_' + storeKey) || '[]');
}
function saveOrdersData(storeKey, orders) {
  localStorage.setItem('mtjri_orders_' + storeKey, JSON.stringify(orders));
}
function getCounter(storeKey) {
  return parseInt(localStorage.getItem('mtjri_counter_' + storeKey) || '0');
}
function incCounter(storeKey) {
  const c = getCounter(storeKey) + 1;
  localStorage.setItem('mtjri_counter_' + storeKey, c);
  return c;
}

let currentStore = null;
let currentStoreKey = null;
let currentFilter = 'all';
let currentSearch = '';
let editingId = null;
let deleteId = null;
let tempImage = null;

// ============ AUTH ============
function switchTab(tab) {
  document.querySelectorAll('.tab-btn').forEach((b,i) => {
    b.classList.toggle('active', (tab==='login' && i===0)||(tab==='register' && i===1));
  });
  document.getElementById('loginForm').style.display = tab==='login' ? 'block' : 'none';
  document.getElementById('registerForm').style.display = tab==='register' ? 'block' : 'none';
  document.getElementById('authError').style.display = 'none';
}

function showError(msg) {
  const el = document.getElementById('authError');
  el.textContent = msg;
  el.style.display = 'block';
}

function makeKey(email) {
  return email.toLowerCase().replace(/[^a-z0-9]/g,'_');
}

function register() {
  const store = document.getElementById('regStore').value.trim();
  const email = document.getElementById('regEmail').value.trim();
  const pass = document.getElementById('regPass').value;
  if (!store || !email || !pass) return showError('الرجاء تعبئة جميع الحقول');
  if (pass.length < 6) return showError('كلمة المرور يجب أن تكون 6 أحرف على الأقل');
  const stores = getStores();
  const key = makeKey(email);
  if (stores[key]) return showError('هذا البريد الإلكتروني مسجل مسبقاً');
  stores[key] = { email, pass, store };
  saveStores(stores);
  showToast('✅ تم إنشاء الحساب بنجاح!');
  switchTab('login');
  document.getElementById('loginEmail').value = email;
}

function login() {
  const email = document.getElementById('loginEmail').value.trim();
  const pass = document.getElementById('loginPass').value;
  if (!email || !pass) return showError('الرجاء إدخال البريد وكلمة المرور');
  const stores = getStores();
  const key = makeKey(email);
  if (!stores[key] || stores[key].pass !== pass) return showError('البريد أو كلمة المرور غير صحيحة');
  currentStore = stores[key];
  currentStoreKey = key;
  localStorage.setItem('mtjri_session', key);
  startApp();
}

function logout() {
  localStorage.removeItem('mtjri_session');
  currentStore = null;
  currentStoreKey = null;
  document.getElementById('appScreen').style.display = 'none';
  document.getElementById('authScreen').style.display = 'flex';
}

function checkSession() {
  const key = localStorage.getItem('mtjri_session');
  if (!key) return;
  const stores = getStores();
  if (stores[key]) {
    currentStore = stores[key];
    currentStoreKey = key;
    startApp();
  }
}

function startApp() {
  document.getElementById('authScreen').style.display = 'none';
  document.getElementById('appScreen').style.display = 'block';
  document.getElementById('storeNameBadge').textContent = '🏪 ' + currentStore.store;
  renderOrders();
}

// ============ ORDERS ============
function getFilteredOrders() {
  let orders = getOrders(currentStoreKey);
  if (currentFilter !== 'all') orders = orders.filter(o => o.status === currentFilter);
  if (currentSearch) {
    const q = currentSearch.toLowerCase();
    orders = orders.filter(o =>
      o.customer.toLowerCase().includes(q) ||
      String(o.num).includes(q) ||
      (o.phone && o.phone.includes(q))
    );
  }
  return orders;
}

function updateStats() {
  const all = getOrders(currentStoreKey);
  document.getElementById('countAll').textContent = all.length;
  document.getElementById('countBuy').textContent = all.filter(o=>o.status==='waiting-buy').length;
  document.getElementById('countDeliver').textContent = all.filter(o=>o.status==='waiting-deliver').length;
  document.getElementById('countShipping').textContent = all.filter(o=>o.status==='shipping').length;
}

function statusLabel(s) {
  if (s==='waiting-buy') return '🛒 انتظار الشراء';
  if (s==='waiting-deliver') return '📫 انتظار التوصيل';
  if (s==='shipping') return '🚚 قيد الشحن';
  return s;
}

function renderOrders() {
  updateStats();
  const orders = getFilteredOrders();
  const el = document.getElementById('ordersList');

  // update filter chips
  document.querySelectorAll('.stat-chip').forEach(c => c.classList.remove('active-filter'));
  const filters = ['all','waiting-buy','waiting-deliver','shipping'];
  const chips = document.querySelectorAll('.stat-chip');
  chips[filters.indexOf(currentFilter)].classList.add('active-filter');

  if (!orders.length) {
    el.innerHTML = `<div class="empty-state">
      <div class="empty-icon">📭</div>
      <h3>لا توجد طلبيات</h3>
      <p>اضغط على "طلب جديد" لإضافة أول طلب</p>
    </div>`;
    return;
  }

  el.innerHTML = orders.slice().reverse().map(o => `
    <div class="order-card" id="card_${o.id}">
      <div class="order-header">
        <div class="order-num">#${o.num}</div>
        <div class="order-status status-${o.status}">${statusLabel(o.status)}</div>
      </div>
      ${o.image ? `<img class="order-img" src="${o.image}" alt="صورة الطلب">` : `<div class="order-img-placeholder">📦</div>`}
      <div class="order-customer">👤 ${o.customer}</div>
      <div class="order-info-row">
        ${o.phone ? `<span class="info-chip">📱 ${o.phone}</span>` : ''}
        ${o.phone2 ? `<span class="info-chip">📲 ${o.phone2}</span>` : ''}
        ${o.size ? `<span class="info-chip">📏 ${o.size}</span>` : ''}
        ${o.social ? `<span class="info-chip">💬 ${o.social}</span>` : ''}
      </div>
      ${o.address ? `<div class="info-chip" style="margin-bottom:8px;display:inline-flex">📍 ${o.address}</div>` : ''}
      ${o.price ? `<div class="order-price">💰 ${Number(o.price).toLocaleString('ar')} د.ع</div>` : ''}
      ${o.tracking ? `<div class="tracking-badge">📦 رقم التتبع: ${o.tracking}</div>` : ''}
      <div class="order-footer">
        <button class="btn-sm btn-edit" onclick="editOrder('${o.id}')">✏️ تعديل</button>
        <button class="btn-sm btn-delete" onclick="openDeleteConfirm('${o.id}')">🗑️ حذف</button>
      </div>
    </div>
  `).join('');
}

function searchOrders(q) {

let imgData = null;

function openModal(id=null) {
  editingId = id;
  imgData = null;
  document.getElementById('modalTitle').textContent = id ? 'تعديل الطلب' : 'طلب جديد';

  if (id) {
    const orders = getOrders(currentStoreKey);
    const o = orders.find(x=>x.id===id);
    if (!o) return;
    document.getElementById('fCustomer').value = o.customer || '';
    document.getElementById('fPhone').value = o.phone || '';
    document.getElementById('fPhone2').value = o.phone2 || '';
    document.getElementById('fAddress').value = o.address || '';
    document.getElementById('fSocial').value = o.social || '';
    document.getElementById('fSize').value = o.size || '';
    document.getElementById('fPrice').value = o.price || '';
    document.getElementById('fTracking').value = o.tracking || '';
    document.querySelectorAll('input[name=status]').forEach(r => r.checked = r.value === o.status);
    statusChanged();
    // image
    const area = document.getElementById('imgUploadArea');
    if (o.image) {
      imgData = o.image;
      let img = area.querySelector('img');
      if (!img) { img = document.createElement('img'); area.appendChild(img); }
      img.src = o.image;
    } else {
      const img = area.querySelector('img');
      if (img) img.remove();
    }
  } else {
    document.getElementById('fCustomer').value = '';
    document.getElementById('fPhone').value = '';
    document.getElementById('fPhone2').value = '';
    document.getElementById('fAddress').value = '';
    document.getElementById('fSocial').value = '';
    document.getElementById('fSize').value = '';
    document.getElementById('fPrice').value = '';
    document.getElementById('fTracking').value = '';
    document.getElementById('s1').checked = true;
    statusChanged();
    const area = document.getElementById('imgUploadArea');
    const img = area.querySelector('img');
    if (img) img.remove();
  }

  document.getElementById('orderModal').classList.add('show');
}

function closeModal() { document.getElementById('orderModal').classList.remove('show'); }
function closeModalOutside(e) { if (e.target === document.getElementById('orderModal')) closeModal(); }

function editOrder(id) { openModal(id); }

function statusChanged() {
  const val = document.querySelector('input[name=status]:checked')?.value;
  document.getElementById('trackingGroup').classList.toggle('show', val === 'shipping');
}

function handleImage(e) {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    imgData = ev.target.result;
    const area = document.getElementById('imgUploadArea');
    let img = area.querySelector('img');
    if (!img) { img = document.createElement('img'); area.appendChild(img); }
    img.src = imgData;
  };
  reader.readAsDataURL(file);
}

function saveOrder() {
  const customer = document.getElementById('fCustomer').value.trim();
  const phone = document.getElementById('fPhone').value.trim();
  const price = document.getElementById('fPrice').value.trim();
  const status = document.querySelector('input[name=status]:checked')?.value;

  if (!customer) return showToast('⚠️ اسم الزبون مطلوب');
  if (!status) return showToast('⚠️ يرجى تحديد حالة الطلب');

  const orders = getOrders(currentStoreKey);

  if (editingId) {
    const idx = orders.findIndex(o => o.id === editingId);
    if (idx !== -1) {
      orders[idx] = {
        ...orders[idx],
        customer,
        phone,
        phone2: document.getElementById('fPhone2').value.trim(),
        address: document.getElementById('fAddress').value.trim(),
        social: document.getElementById('fSocial').value.trim(),
        size: document.getElementById('fSize').value.trim(),
        price,
        status,
        tracking: status === 'shipping' ? document.getElementById('fTracking').value.trim() : '',
        image: imgData || orders[idx].image || null,
        updatedAt: Date.now()
      };
    }
    showToast('✅ تم تعديل الطلب');
  } else {
    const num = incCounter(currentStoreKey);
    orders.push({
      id: 'o_' + Date.now(),
      num,
      customer,
      phone,
      phone2: document.getElementById('fPhone2').value.trim(),
      address: document.getElementById('fAddress').value.trim(),
      social: document.getElementById('fSocial').value.trim(),
      size: document.getElementById('fSize').value.trim(),
      price,
      status,
      tracking: status === 'shipping' ? document.getElementById('fTracking').value.trim() : '',
      image: imgData || null,
      createdAt: Date.now()
    });
    showToast('✅ تم إضافة الطلب بنجاح');
  }

  saveOrdersData(currentStoreKey, orders);
  closeModal();
  renderOrders();
}

function closeConfirm() {
  deleteId = null;

function confirmDelete() {

  let orders = getOrders(currentStoreKey);
  orders = orders.filter(o => o.id !== deleteId);
  saveOrdersData(currentStoreKey, orders);
  closeConfirm();
  renderOrders();
  showToast('🗑️ تم حذف الطلب');
}

}
